atribuicao:
	<IDENTIFICADO,> <IGUAL,> (expressão)* + expressao <PONTOVIRGULA,>

exibicao:
	<OUT,> (expressão <VIRGULA,>)* (expressão) <PONTOVIRGULA,> 

entrada:
	<IN,> (<IDENTIFICADO,> <VIRGULA,>)* <IDENTIFICADO,>  <PONTOVIRGULA,> 
	
condicional:
	<IF,> <LPARENTES,> expressão <RPARENTES,> <LCHAVES,> Comando <RCHAVES,> CondElse

expressão:
	ehparentes ehExpressao ehOperador ehExpressao ehparentes
	ehExpressao ehOperador ehExpressao 
 	ehVariavelBoleano not

EXPRESSÃO:: // Outro modo de ver
	E0 -> B = E1 | E1
	E1 -> E1 C OPERADORES D | D
	C -> not | vazio
	D -> B | ( E1 )
	B -> X 


ehparentes:
	<LPARENTES,> || vazio

ehVariavelBoleano
	<IDENTIFICADOR,> || ehBoleano 

not:
	ehNot || vazio

	
CondElse:
	<ELSE,> <LCHAVES,> condicional comando <RCHAVES,> || 
	<ELSE,> <LCHAVES,> comando <RCHAVES,> 
	
comando:
	expressão comando || atribuicao comando || atribuicao || expressao || vazia
